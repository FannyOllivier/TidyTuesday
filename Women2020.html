# transformer du texte en corpus qui pourra tre trait
text_corpus <- Corpus(VectorSource(women$description))
# enlever les majuscules
#text_corpus <- tm_map(text_corpus, content_transformer(tolower))
text_corpus <- tm_map(text_corpus, removePunctuation)
## Warning in tm_map.SimpleCorpus(text_corpus, removePunctuation): transformation
## drops documents
text_corpus <- tm_map(text_corpus, function(x)removeWords(x,stopwords("english"))) #stopwords(kind = "fr") pour supprimer les mots les plus courants en franais (le, la, les ...), par dfaut en anglais
## Warning in tm_map.SimpleCorpus(text_corpus, function(x) removeWords(x,
## stopwords("english"))): transformation drops documents
set.seed(1234)

wordcloud(text_corpus,min.freq = 3,scale = c(3, 0.3),
          max.word=Inf,colors=brewer.pal(4,"Dark2"),
          random.order = FALSE,random.color = TRUE)
